35 int DSPL_API random_init(random_t* prnd, int type, void* seed)
36 {
37  srand(time(NULL));
38 
39  if(!prnd)
40  return RES_OK;
41 
42  switch(type)
43  {
44  case RAND_TYPE_MRG32K3A:
45  /* MRG32k3a init */
46  prnd->mrg32k3a_x[0] = prnd->mrg32k3a_x[1] = 1.0;
47  prnd->mrg32k3a_y[0] = prnd->mrg32k3a_y[1] = prnd->mrg32k3a_y[2] = 1.0;
48  if(seed)
49  prnd->mrg32k3a_x[2] = *((double*)seed);
50  else
51  prnd->mrg32k3a_x[2] = (double) rand() * rand();
52  break;
53  case RAND_TYPE_MT19937:
54  if(seed)
55  mt19937_init_genrand64(*((unsigned long long*)seed), prnd);
56  else
57  mt19937_init_genrand64((unsigned long long)rand()*rand(), prnd);
58  break;
59  default:
60  return ERROR_RAND_TYPE;
61  }
62  prnd->type = type;
63 
64  return RES_OK;
65 }

70 /******************************************************************************
71 Uniform random generator mrg32k3a
72 *******************************************************************************/
73 int randu_mrg32k3a (double* u, int n, random_t* prnd)
74 {
75 
76  if(!u || !prnd)
77  return ERROR_PTR;
78  if(n < 1)
79  return ERROR_SIZE;
80 
81  long z;
82  double xn, yn, *x, *y;
83  int k;
84 
85  x = prnd->mrg32k3a_x;
86  y = prnd->mrg32k3a_y;
87  for(k = 0; k < n; k++)
88  {
89  /* Component x[n] */
90  xn = MRG32K3A_A12 * x[1] - MRG32K3A_A13 * x[2];
91 
92  z = (long)(xn / MRG32K3A_M1);
93  xn -= (double)z * MRG32K3A_M1;
94  if (xn < 0.0)
95  xn += MRG32K3A_M1;
96 
97  x[2] = x[1];
98  x[1] = x[0];
99  x[0] = xn;
100 
101  /* Component y[n] */
102  yn = MRG32K3A_A21 * y[0] - MRG32K3A_A23 * y[2];
103  z = (long)(yn / MRG32K3A_M2);
104  yn -= (double)z * MRG32K3A_M2;
105  if (yn < 0.0)
106  yn += MRG32K3A_M2;
107 
108  y[2] = y[1];
109  y[1] = y[0];
110  y[0] = yn;
111 
112  /* Combination */
113  u[k] = (xn <= yn) ? ((xn - yn + MRG32K3A_M1) * MRG32K3A_NORM):
114  (xn - yn) * MRG32K3A_NORM;
115  }
116  return RES_OK;
117 }

126 Uniform random numbers generator
127 *******************************************************************************/
128 int DSPL_API randu(double* x, int n, random_t* prnd)
129 {
130  int i;
131 
132  if(!x)
133  return ERROR_PTR;
134  if(n < 0)
135  return ERROR_SIZE;
136 
137  if(prnd)
138  {
139  switch(prnd->type)
140  {
141  case RAND_TYPE_MRG32K3A:
142  return randu_mrg32k3a(x, n, prnd);
143  case RAND_TYPE_MT19937:
144  for(i = 0; i < n; i++)
145  x[i] = mt19937_genrand64_real1(prnd);
146  return RES_OK;
147  default:
148  return ERROR_RAND_TYPE;
149  }
150  }
151  else
152  {
153  if(!x)
154  return ERROR_PTR;
155  if(n<1)
156  return ERROR_SIZE;
157  for(i = 0; i < n; i++)
158  x[i] = (double)rand()/RAND_MAX;
159  }
160 
161  return RES_OK;
162 }

167 /*******************************************************************************
168 Gaussian random numbers generator
169 *******************************************************************************/
170 int DSPL_API randn(double* x, int n, double mu, double sigma, random_t* prnd)
171 {
172  int k, m;
173  double x1[512], x2[512];
174  int res;
175  if(!x)
176  return ERROR_PTR;
177 
178  if(n<1)
179  return ERROR_SIZE;
180 
181  if(sigma < 0.0)
182  return ERROR_RAND_SIGMA;
183 
184  k=0;
185  while(k < n)
186  {
187  if((res = randu(x1, 512, prnd)) != RES_OK)
188  goto exit_label;
189  if((res = randu(x2, 512, prnd)) != RES_OK)
190  goto exit_label;
191  m = 0;
192  while(k < n && m < 512)
193  {
194  if(x1[m] != 0.0)
195  {
196  x[k] = sqrt(-2.0*log(x1[m]))*cos(M_2PI*x2[m])*sigma + mu;
197  k++;
198  m++;
199  }
200  }
201  }
202 
203  res = RES_OK;
204 exit_label:
205  return res;
206 }
