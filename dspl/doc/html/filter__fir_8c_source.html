31 int fir_linphase_lpf(int ord, double wp, int wintype,
32  double winparam, double* h)
33 {
34  int n, err = RES_OK;
35  double *w = NULL;
36 
37 
38  w = (double*)malloc((ord+1)*sizeof(double));
39 
40  err = linspace(-(double)ord*0.5, (double)ord*0.5, ord+1, DSPL_SYMMETRIC, w);
41 
42  if(err!=RES_OK)
43  goto error_proc;

67 /******************************************************************************
68  * Linear phase FIR filter
69  ******************************************************************************/
70 int DSPL_API fir_linphase(int ord, double w0, double w1, int filter_type,
71  int wintype, double winparam, double* h)
72 {
73  int n, err;
74  double wc, b, del;
75 
76  if(ord<1)
77  return ERROR_FILTER_ORD;
78  if(w0 <= 0.0)
79  return ERROR_FILTER_WP;
80  if(!h)
81  return ERROR_PTR;
82 
83  switch(filter_type & DSPL_FILTER_TYPE_MASK)
84  {
85  /* Lowpass FIR coefficients calculation */
86  case DSPL_FILTER_LPF:
87  err = fir_linphase_lpf(ord, w0, wintype, winparam, h);
88  break;
89 
90  /* Highpass FIR coefficients calculation */
91  case DSPL_FILTER_HPF:
92  err = fir_linphase_lpf(ord, 1.0-w0, wintype, winparam, h);
93  if(err == RES_OK)
94  {
95  /* LPF filter frequency inversion */
96  for(n = 0; n < ord+1; n+=2)
97  h[n] = -h[n];
98  }
99  break;
100 
101  /* Bandpass FIR coefficients calculation */
102  case DSPL_FILTER_BPASS:
103  if(w1 < w0)
104  {
105  err = ERROR_FILTER_WS;
106  break;
107  }
108  wc = (w0 + w1) * 0.5; /* central frequency */
109  b = w1 - w0; /* bandwidth */
110  err = fir_linphase_lpf(ord, b*0.5, wintype, winparam, h);
111  if(err == RES_OK)
112  {
113  /* LPF frequency shifting to the central frequency */
114  del = 0.5 * (double)ord;
115  for(n = 0; n < ord+1; n++)
116  h[n] *= 2.0 * cos(M_PI * ((double)n - del) * wc);
117  }
118  break;
119 
120  /* BandStop FIR coefficients calculation */
121  /* ATTENTION! Bandstop filter must be even order only! */
122  case DSPL_FILTER_BSTOP:
123  {
124  double *h0 = NULL;
125 
126  /* check filter order. Return error if order is odd. */
127  if(ord%2)
128  return ERROR_FILTER_ORD;
129 
130  /* check frequency (w1 must be higher than w0) */
131  if(w1 < w0)
132  {
133  err = ERROR_FILTER_WS;
134  break;
135  }
136  /* temp coeff vector */
137  h0 = (double*)malloc((ord+1) * sizeof(double));
138 
139  /* calculate LPF */
140  err = fir_linphase(ord, w0, 0.0, DSPL_FILTER_LPF, wintype, winparam, h0);
141  if(err!=RES_OK)
142  {
143  free(h0);
144  return err;
145  }
146  /* calculate HPF */
147  err = fir_linphase(ord, w1, 0.0, DSPL_FILTER_HPF, wintype, winparam, h);
148  if(err==RES_OK)
149  {
150  /* Bandstop filter is sum of lowpass and highpass filters */
151  for(n = 0; n < ord+1; n++)
152  h[n] += h0[n];
153  }
154  free(h0);
155  break;
156  }
157  default:
158  err = ERROR_FILTER_FT;
159  }
160  return err;
161 }
