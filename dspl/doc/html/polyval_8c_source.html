29 Polynomial zeros to coefficients

31 int DSPL_API poly_z2a_cmplx(complex_t* z, int nz, int ord, complex_t* a)
32 {
33  int k, ind, res;
34  complex_t x[2];
35 
36  if(!z || !a)
37  return ERROR_PTR;
38  if(nz < 0)
39  return ERROR_SIZE;
40  if(nz > ord || ord < 1)
41  return ERROR_POLY_ORD;
42 
43  RE(x[1]) = 1.0;
44  IM(x[1]) = 0.0;
45 
46  memset(a, 0, (ord+1) * sizeof(complex_t));
47 
48  RE(a[0]) = 1.0;
49  ind = 1;
50  for(k = 0; k < nz; k++)
51  {
52  RE(x[0]) = -RE(z[k]);
53  IM(x[0]) = -IM(z[k]);
54  res = conv_cmplx(a, ind, x, 2, a);
55  if(res!=RES_OK)
56  return res;
57  ind++;
58  }
59 
60  return RES_OK;
61 }

66 /******************************************************************************
67 Real polynomial evaluation
68 *******************************************************************************/
69 int DSPL_API polyval(double* a, int ord, double* x, int n, double* y)
70 {
71  int k, m;
72 
73  if(!a || !x || !y)
74  return ERROR_PTR;
75  if(ord<0)
76  return ERROR_POLY_ORD;
77  if(n<1)
78  return ERROR_SIZE;
79 
80  for(k = 0; k < n; k++)
81  {
82  y[k] = a[ord];
83  for(m = ord-1; m>-1; m--)
84  y[k] = y[k]*x[k] + a[m];
85  }
86  return RES_OK;
87 }

93 /*******************************************************************************
94 Complex polynomial evaluation
95 *******************************************************************************/
96 int DSPL_API polyval_cmplx(complex_t* a, int ord,
97  complex_t* x, int n, complex_t* y)
98 {
99  int k, m;
100  complex_t t;
101 
102  if(!a || !x || !y)
103  return ERROR_PTR;
104  if(ord<0)
105  return ERROR_POLY_ORD;
106  if(n<1)
107  return ERROR_SIZE;
108 
109  for(k = 0; k < n; k++)
110  {
111  RE(y[k]) = RE(a[ord]);
112  IM(y[k]) = IM(a[ord]);
113  for(m = ord-1; m>-1; m--)
114  {
115  RE(t) = CMRE(y[k], x[k]);
116  IM(t) = CMIM(y[k], x[k]);
117  RE(y[k]) = RE(t) + RE(a[m]);
118  IM(y[k]) = IM(t) + IM(a[m]);
119  }
120  }
121  return RES_OK;
122 }
